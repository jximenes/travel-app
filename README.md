# travel-app
	npm install bower -g
	npm install sass -g
	npm install http-server -g

## install bower dependencies (angularjs)
	bower install
	
## run sass pre-processor
	sass scss/main.scss assets/css/stylesheets.css

## run web server
	http-server -o
